PROGRAM         : FLOOR_INIT STATEMENT_LIST | STATEMENT_LIST

FLOOR_INIT      : floor size INT

STATEMENT_LIST  : STATEMENT STATEMENT_LIST | STATEMENT

STATEMENT       : LET | WHILE | ASSIGN
LET             : let ID
WHILE           : while CONDITION BLOCK
ASSIGN          : WRITE_REF = READ_EXP

CONDITION       : true
BLOCK           : { STATEMENT_LIST }

READ_EXP        : ADD_EXP | READ_REF
ADD_EXP         : READ_REF +/- READ_REF

READ_REF        : INBOX | ID | FLOOR_SLOT
WRITE_REF       : OUTBOX | HAND | ID

INBOX           : inbox
OUTBOX          : outbox
FLOOR_SLOT      : floor [ POS_INT ]
ID              : [a-zA-z]+

VALUE_LIST      : VALUE , VALUE | VALUE
VALUE           : LETTER | INT
LETTER          : [A-Z]
INT             : POS_INT | NEG_INT
NEG_INT         : -[1-9][0-9]*
POS_INT         : [1-9][0-9]* | 0
